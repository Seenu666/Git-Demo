data(AirPassengers)
APTS<-ts(AirPassengers, frequency=12, start = c(1949,01))
head(APTS)
APTS_decompose =decompose(APTS)
plot(APTS_decompose)
plot.ts(APTS)
install.packages(TTR)
library(TTR)
sm=SMA(APTS,n=8)
plot(sm)
APTSfOrecasts<-HoltWinters(APTS, alpha = 0.9, beta = FALSE, gamma = FALSE)
plot(APTSfOrecasts)
install.packages(forecast)
library(forecast)
Forecast<- forecast(APTSfOrecasts,h=3)
Forecast
plot(Forecast)
plot.ts(Forecast)
